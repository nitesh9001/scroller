## Install

```bash
npm install --save scroller
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'scroller'
import 'scroller/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
